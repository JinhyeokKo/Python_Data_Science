np.random.seed(42)
x = np.random.randn(100)

#   
bins = np.linspace(-5, 5, 20)
counts = np.zeros_like(bins)

#  x    
i = np.searchsorted(bins, x)

#   1 
np.add.at(counts, i, 1)

plt.plot(bins, counts, drawstyle='steps');
